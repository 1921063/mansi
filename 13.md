## Program to swap two numbers.


#include <stdio.h>

int  main()  
{  
int  x,  y,  t;

printf("Enter two integers\n");  
scanf("%d%d",  &x,  &y);

printf("Before Swapping\nFirst integer = %d\nSecond integer = %d\n",  x,  y);

t  =  x;  
x  =  y;  
y  =  t;

printf("After Swapping\nFirst integer = %d\nSecond integer = %d\n",  x,  y);

return  0;  
}


## output

Enter two integers  
23  
45  
Before Swapping  
First integer = 23  
Second integer = 45  
After Swapping  
First integer = 45  
Second integer = 23

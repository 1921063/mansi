## Program to display multiplication table of the given range of numbers.


#include<stdio.h>
 int main()
 {
 int x, i, t, a, b, n;
 printf(“\n \n Display table from a to b:   ”);
scanf(“%d%d”, &a , &b);
 printf(“\n \nHow many multiples should be displayed : “);
scanf(“%d”, &n);
printf(“\n \n”);
For(i=1;i<=n;i++)
{
 For(x=a;x<=b;x++)
{
t=x*1;
Printf(“%d X %d = %d \t”, x, i, t);
 }
printf(“\n”);
}
Return 0;
 }
